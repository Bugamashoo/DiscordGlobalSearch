(() => {
  // ================================================================
  // FEATURE 4: Fetch + XHR interception for auth token and API version
  // ================================================================
  // Patch fetch and XHR immediately so we passively capture the
  // Authorization header and API version from any outgoing Discord API
  // request. Both patches are fully transparent and forward all calls to
  // the originals. Restores itself once both values are captured.
  let _capturedToken = null;
  let _capturedApiVersion = null;
  const _origFetch = window.fetch;
  const _origXHROpen = XMLHttpRequest.prototype.open;
  const _origXHRSetHeader = XMLHttpRequest.prototype.setRequestHeader;
  const _xhrUrls = new WeakMap();

  const _netCapture = (urlStr, authHeader) => {
    if (!urlStr || !urlStr.includes("discord.com/api/")) return;
    if (!_capturedApiVersion) {
      const m = urlStr.match(/discord(?:app)?\.com\/api\/(v\d+)\//);
      if (m) _capturedApiVersion = m[1];
    }
    if (!_capturedToken && typeof authHeader === "string" && authHeader.length > 30)
      _capturedToken = authHeader;
    if (_capturedToken && _capturedApiVersion) {
      window.fetch = _origFetch;
      XMLHttpRequest.prototype.open = _origXHROpen;
      XMLHttpRequest.prototype.setRequestHeader = _origXHRSetHeader;
    }
  };

  window.fetch = function(url, init) {
    try {
      const urlStr = String(typeof url === "string" ? url : (url?.url ?? url));
      let auth;
      if (init?.headers instanceof Headers) auth = init.headers.get("authorization");
      else if (init?.headers) auth = init.headers["authorization"] ?? init.headers["Authorization"];
      _netCapture(urlStr, auth);
    } catch (e) {}
    return _origFetch.apply(this, arguments);
  };

  XMLHttpRequest.prototype.open = function(method, url) {
    try { _xhrUrls.set(this, String(url)); } catch (e) {}
    return _origXHROpen.apply(this, arguments);
  };

  XMLHttpRequest.prototype.setRequestHeader = function(name, value) {
    try {
      if (name.toLowerCase() === "authorization")
        _netCapture(_xhrUrls.get(this) ?? "", value);
    } catch (e) {}
    return _origXHRSetHeader.apply(this, arguments);
  };

  // ================================================================
  // Webpack chunk discovery: try the known name first, then scan window
  // for any property matching the webpackChunk* pattern in case Discord
  // renames the app bundle
  const getWebpackChunk = () => {
    if (window.webpackChunkdiscord_app) return window.webpackChunkdiscord_app;
    for (const key of Object.getOwnPropertyNames(window)) {
      try {
        if (key.startsWith("webpackChunk") && Array.isArray(window[key]) && window[key].push) {
          return window[key];
        }
      } catch (e) {}
    }
    return null;
  };

  let _wpRequire = null;
  const wp = () => {
    if (_wpRequire) return _wpRequire;
    const chunk = getWebpackChunk();
    if (!chunk) return null;
    _wpRequire = chunk.push([[Symbol()], {}, r => r]);
    chunk.pop();
    return _wpRequire;
  };

  // Extract all possible export surfaces from a webpack module.
  // Checks the top level, the default export, and all object valued
  // keys so renamed or restructured modules still surface.
  const candidatesOf = (mod) => {
    if (!mod?.exports) return [];
    const seen = new Set();
    const out = [];
    const add = (v) => {
      if (v && typeof v === "object" && !seen.has(v)) {
        seen.add(v);
        out.push(v);
      }
    };
    add(mod.exports);
    try {
      if (mod.exports.default) add(mod.exports.default);
      for (const k of Object.keys(mod.exports)) {
        add(mod.exports[k]);
      }
    } catch (e) {}
    return out;
  };

  // Discord's i18n system uses Proxy objects where every property access
  // returns a locale message getter function. These proxies match filters
  // for transitionTo, getToken, getGuilds, etc. because they respond to
  // ANY property name with a function. Detect them by probing a property
  // that could never exist on a real module: if it returns a function,
  // the object is an i18n proxy and should be skipped.
  const isI18nProxy = (obj) => {
    try {
      return typeof obj["__xsearch_i18n_probe__"] === "function";
    } catch (e) {
      return false;
    }
  };

  // Build and cache a flat list of every exported object across all
  // webpack modules, excluding i18n proxies. Cached because the module
  // graph is static at runtime.
  const _buildCandidates = () => {
    const wpRequire = wp();
    if (!wpRequire || !wpRequire.c) return [];
    const out = [];
    for (const id in wpRequire.c) {
      try {
        for (const exp of candidatesOf(wpRequire.c[id])) {
          if (!isI18nProxy(exp)) out.push(exp);
        }
      } catch (e) {}
    }
    return out;
  };
  const _candidates = _buildCandidates();

  // ================================================================
  // FEATURE 1: Dynamic API base URL detection
  // ================================================================
  // Detect the live API version instead of hardcoding v9. Priority order:
  // 1. Already captured from a live request (most authoritative)
  // 2. window.GLOBAL_ENV.API_ENDPOINT exposed by Discord's client build
  // 3. Webpack module export containing an API URL string with a version
  // 4. Hardcoded fallback
  const getApiBase = () => {
    if (_capturedApiVersion) return `https://discord.com/api/${_capturedApiVersion}`;
    try {
      const env = window.GLOBAL_ENV;
      if (env?.API_ENDPOINT) {
        const ep = env.API_ENDPOINT.replace(/\/$/, "");
        if (/\/v\d+$/.test(ep)) return ep;
        if (env.API_VERSION != null) return `${ep}/v${env.API_VERSION}`;
      }
    } catch (e) {}
    for (const exp of _candidates) {
      try {
        for (const key of Object.keys(exp)) {
          const val = exp[key];
          if (typeof val === "string" && /^https:\/\/discord(?:app)?\.com\/api\/v\d+$/.test(val))
            return val;
        }
      } catch (e) {}
    }
    return "https://discord.com/api/v9";
  };
  const API_BASE = getApiBase();

  // TOKEN HANDLING: the token is read from Discord's own internal stores
  // (the same way Discord itself reads it) and is ONLY ever sent to
  // discord.com in the authorization header of search requests.
  // It is never logged, stored externally, or transmitted anywhere else.
  // You can verify this by searching this entire script for "token":
  // it only appears in this function, the variable assignment below,
  // and the fetch() call in the search loop.
  const jwtLike = /^[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{4,}\.[A-Za-z0-9_-]{10,}$/;
  const isTokenLike = (t) => typeof t === "string" && t.length > 30 && jwtLike.test(t);

  const getToken = () => {
    // Strategy 1: network capture (set by the fetch/XHR patch above; most resilient)
    if (_capturedToken) return _capturedToken;
    // Strategy 2: read from Discord's own webpack auth store (kept in memory)
    for (const exp of _candidates) {
      try {
        if (typeof exp.getToken !== "function") continue;
        const t = exp.getToken();
        if (isTokenLike(t)) return t;
      } catch (e) {}
    }
    // Strategy 3: read from localStorage (older builds stored it here)
    try {
      const raw = localStorage.getItem("token");
      if (raw) {
        const cleaned = raw.replace(/^"|"$/g, "");
        if (isTokenLike(cleaned)) return cleaned;
      }
    } catch (e) {}
    // Strategy 4 (FEATURE 3): schema-based structural scan
    // Call every zero-argument function on modules that look like store modules
    // (2 to 30 methods, to avoid large dispatch modules with side effects).
    // A JWT-shaped return value is highly distinctive, so false positives are rare.
    for (const exp of _candidates) {
      try {
        const fns = Object.keys(exp).filter(k => typeof exp[k] === "function");
        if (fns.length < 2 || fns.length > 30) continue;
        for (const key of fns) {
          const t = exp[key]();
          if (isTokenLike(t)) return t;
        }
      } catch (e) {}
    }
    return null;
  };

  // Guilds: validates that the returned value is a map of guild-shaped objects
  // using guild-specific fields (features, roles, ownerId) so other id/name maps
  // from unrelated modules are not mistaken for guild data.
  const looksLikeGuild = (v) =>
    v && typeof v.id === "string" && typeof v.name === "string"
    && (Array.isArray(v.features) || Array.isArray(v.roles) || v.ownerId != null);

  const isGuildMap = (obj) => {
    if (!obj || typeof obj !== "object" || Array.isArray(obj)) return false;
    const vals = Object.values(obj);
    if (vals.length === 0) return false;
    return vals.slice(0, Math.min(3, vals.length)).every(looksLikeGuild);
  };

  const extractGuildList = (obj) =>
    Object.values(obj)
      .filter(v => v?.id && v?.name)
      .map(v => ({ id: v.id, name: v.name }));

  const getGuilds = () => {
    // Strategy 1: name-based (fast path)
    for (const exp of _candidates) {
      try {
        if (typeof exp.getGuilds !== "function") continue;
        const g = exp.getGuilds();
        if (isGuildMap(g)) return extractGuildList(g);
      } catch (e) {}
    }
    // Strategy 2 (FEATURE 3): schema-based structural scan
    // Call every function on every module and check if the return value is
    // a map of guild-shaped objects. Guild-specific fields (features, roles,
    // ownerId) prevent false matches against other id/name maps.
    for (const exp of _candidates) {
      try {
        for (const key of Object.keys(exp)) {
          if (typeof exp[key] !== "function") continue;
          const result = exp[key]();
          if (isGuildMap(result)) return extractGuildList(result);
        }
      } catch (e) {}
    }
    return [];
  };

  // Generic module finder for navigation lookups
  const findModule = (filter) => {
    for (const exp of _candidates) {
      try { if (filter(exp)) return exp; } catch (e) {}
    }
    return null;
  };

  // Navigation: three fallback strategies so renames of any single
  // internal module do not break message jumping.
  // Strategy 1: transitionTo (Router module, most common)
  // Strategy 2: selectChannel (ChannelActions module)
  // Strategy 3: history.pushState + popstate (no webpack module needed)
  const getTransitionTo = () => {
    // Strategy 1: name-based with paired method verification
    const mod =
      findModule(m => typeof m.transitionTo === "function" && typeof m.replaceWith === "function")
      ?? findModule(m => typeof m.transitionTo === "function");
    if (mod) return mod.transitionTo.bind(mod);
    // Strategy 2 (FEATURE 3): schema-based router detection
    // React Router exposes a history object with push/replace and a location
    // with pathname. This shape is distinctive enough to match without names.
    const routerMod = findModule(m =>
      m?.history && typeof m.history.push === "function"
      && typeof m.history.replace === "function"
      && m?.location && typeof m.location.pathname === "string"
    );
    if (routerMod) return (path) => routerMod.history.push(path);
    return null;
  };

  const getSelectChannel = () => {
    // Strategy 1: name-based with paired method verification
    const mod =
      findModule(m => typeof m.selectChannel === "function" && typeof m.selectVoiceChannel === "function")
      ?? findModule(m => typeof m.selectChannel === "function");
    if (mod) return mod.selectChannel.bind(mod);
    // Strategy 2 (FEATURE 3): find a channel-actions module by characteristic
    // sibling methods (selectVoiceChannel, selectPrivateChannel) even when
    // selectChannel itself has been renamed by minification.
    const chanMod = findModule(m =>
      typeof m.selectVoiceChannel === "function" || typeof m.selectPrivateChannel === "function"
    );
    if (chanMod) {
      const fn = chanMod.selectChannel ?? chanMod.selectPrivateChannel ?? chanMod.selectVoiceChannel;
      if (typeof fn === "function") return fn.bind(chanMod);
    }
    return null;
  };

  const transitionTo = getTransitionTo();
  const selectChannel = getSelectChannel();

  const jumpToMessage = (guildId, channelId, messageId) => {
    const path = `/channels/${guildId}/${channelId}/${messageId}`;
    try { if (transitionTo) { transitionTo(path); return; } } catch (e) {}
    try { if (selectChannel) { selectChannel({ guildId, channelId, messageId }); return; } } catch (e) {}
    // Fallback: direct history manipulation (no webpack module needed)
    try {
      window.history.pushState({}, "", path);
      window.dispatchEvent(new PopStateEvent("popstate"));
    } catch (e) { console.error("[xsearch] no navigation method worked", e); }
  };

  // Init
  // Token is stored in a local variable scoped to this IIFE. It never
  // leaves this closure and is only used in the authorization header
  // of fetch() calls to discord.com (see the search function below).
  let token = getToken();

  const guilds = getGuilds().sort((a, b) => a.name.localeCompare(b.name));
  if (guilds.length === 0) { console.error("[xsearch] Found 0 guilds."); return; }
  if (!token) console.warn("[xsearch] Token not found via webpack; will retry at search time via network capture.");
  console.log(`[xsearch] Ready: ${guilds.length} guilds, nav=${!!(transitionTo || selectChannel)}, api=${API_BASE}`);

  // ================================================================
  // FEATURE 5: CSP-resilient style injection

      document.adoptedStyleSheets = [...(document.adoptedStyleSheets ?? []), sheet];

  };

  // Overlay UI
  document.getElementById("xsearch-overlay")?.remove();
  _removeStyles?.();

  const search = async (query) => {
    // Re-check token here; it may have been captured from the network after init
    if (!token) token = getToken();
    const checked = [...serversDiv.querySelectorAll("input:checked")].map(i => ({

      name: i.parentElement.querySelector("span").textContent.trim()
    }));

    const status = overlay.querySelector("#xsearch-status");
    const results = overlay.querySelector("#xsearch-results");

    const progressBar = overlay.querySelector("#xsearch-progress-bar");

    progressBar.style.width = "0%";
    exportBtn.disabled = true;
    goBtn.disabled = true;
    stopBtn.disabled = false;
    stopRequested = false;
    lastResults = [];

    if (!token) {
      status.textContent = "Error: could not retrieve auth token. Try navigating to a channel first.";
      goBtn.disabled = false;
      return;
    }
    if (checked.length === 0) { alert("Select at least one server."); return; }

    const all = [];
    const total = checked.length;

    // Progress model: each server gets an equal 1/total slice of the bar.
    // Pages within a server subdivide that slice. A 0 result server
    // completes its full slice immediately. The bar only moves forward
    // because a server with many pages just fills its own slice more
    // slowly; it never reduces progress already shown.
    let completedServers = 0;
    const updateProgress = (serverFraction) => {
      const pct = ((completedServers + Math.min(1, serverFraction)) / total) * 100;
      progressBar.style.width = `${Math.min(100, pct)}%`;
    };

    outer:
    for (let s = 0; s < total; s++) {
      const g = checked[s];
      let offset = 0;
      let serverTotal = null;
      let serverPage = 1;
      let serverPages = 1;
      let indexRetries = 0;

      updateProgress(0);

      while (true) {
        if (stopRequested) break outer;
        const pageInfo = serverTotal != null
          ? `page ${serverPage} of ~${Math.ceil(serverTotal / PAGE_SIZE)}`
          : `page ${serverPage}`;
        status.textContent = `Server ${s + 1}/${total}: ${g.name} - ${pageInfo} (${all.length} results so far)`;

        try {
          // NETWORK REQUEST: search sent to discord.com only (never any third party)
          // This is the ONLY place the token is used after retrieval
          const url = `${API_BASE}/guilds/${g.id}/messages/search?content=${encodeURIComponent(query)}&offset=${offset}`;
          const res = await fetch(url, { headers: { authorization: token } });

          // ================================================================
          // FEATURE 2: Parse Retry-After header/body instead of hardcoding 20s
          // ================================================================
          if (res.status === 429) {
            let retryMs = 20000;
            try {
              const headerVal = res.headers.get("Retry-After");
              if (headerVal != null) {
                retryMs = parseFloat(headerVal) * 1000;
              } else {
                const body = await res.json();
                if (typeof body.retry_after === "number") retryMs = body.retry_after * 1000;
              }
            } catch (e) {}
            const totalSecs = Math.max(1, Math.ceil(retryMs / 1000));
            for (let remaining = totalSecs; remaining > 0; remaining--) {
              if (stopRequested) break outer;
              status.textContent = `Rate limited (429) on ${g.name} page ${serverPage} - retrying in ${remaining}s...`;
              await sleep(1000);
            }
            continue;
          }

          // Handle 202: Discord returns this when the server is still
          // being indexed and search results are not yet available
          if (res.status === 202) {
            indexRetries++;
            if (indexRetries > MAX_INDEX_RETRIES) {
              console.warn(`[xsearch] ${g.name} still indexing after ${MAX_INDEX_RETRIES} retries, skipping`);
              status.textContent = `${g.name} is still indexing, skipping... (${all.length} results so far)`;
              break;
            }
            status.textContent = `${g.name} is being indexed by Discord, waiting... (attempt ${indexRetries}/${MAX_INDEX_RETRIES})`;
            await sleep(INDEX_WAIT);
            continue;
          }

          if (!res.ok) {
            console.warn(`[xsearch] ${g.name} returned ${res.status}`);
            break;
          }

          const data = await res.json();

          // Some Discord versions return 200 with an indexing flag in the
          // body instead of a 202 status. Detect this by checking for the
          // flag or for a suspiciously empty first page with an analytics_id
          // (which indicates the server acknowledged the query but has no
          // index to search yet).
          const bodyIndicatesIndexing = data.indexing
            || (offset === 0 && data.total_results === 0 && data.analytics_id && !(data.messages?.length));
          if (bodyIndicatesIndexing) {
            indexRetries++;
            if (indexRetries > MAX_INDEX_RETRIES) {
              console.warn(`[xsearch] ${g.name} still indexing after ${MAX_INDEX_RETRIES} retries, skipping`);
              status.textContent = `${g.name} is still indexing, skipping... (${all.length} results so far)`;
              break;
            }
            status.textContent = `${g.name} is being indexed by Discord, waiting... (attempt ${indexRetries}/${MAX_INDEX_RETRIES})`;
            await sleep(INDEX_WAIT);
            continue;
          }

          // Reset indexing counter on a successful real response
          indexRetries = 0;

          if (serverTotal == null) {
            serverTotal = data.total_results || 0;
            serverPages = Math.max(1, Math.ceil(serverTotal / PAGE_SIZE));
          }
          const groups = data.messages || [];
          if (groups.length === 0) break;

          for (const group of groups) {
            const hit = group.find(m => m.hit) || group[0];
            if (hit) {
              const enriched = { ...hit, _guildId: g.id, _guildName: g.name };
              all.push(enriched);
            }
          }

          updateProgress(serverPage / serverPages);

          offset += PAGE_SIZE;
          if (offset >= serverTotal || groups.length < PAGE_SIZE) break;
          serverPage++;

          await sleep(PAGE_DELAY);
        } catch (e) {
          console.error(`[xsearch] ${g.name}`, e);
          break;
        }
      }

      // Server finished (0 results, all pages done, error, or skip).
      // Advance the bar by completing this server's full slice.
      completedServers++;
      updateProgress(0);

      if (s < total - 1 && !stopRequested) {
        status.textContent = `Waiting before next server... (${all.length} results so far)`;
        await sleep(SERVER_DELAY);
      }
    }

    all.sort((a, b) => new Date(a.timestamp) - new Date(b.timestamp));
    lastResults = all;
    status.textContent = `${all.length} total results across ${total} servers${stopRequested ? " (stopped early)" : ""}`;
    progressWrap.style.display = "none";
    progressBar.style.width = "100%";
    exportBtn.disabled = all.length === 0;
    goBtn.disabled = false;
    stopBtn.disabled = true;

    for (const m of all) renderResult(m);
  };

  // CSV export (local file download only, no network requests)
  const csvEscape = (v) => {
    if (v == null) return "";
    const s = String(v);
    if (/[",\n\r]/.test(s)) return `"${s.replace(/"/g, '""')}"`;
    return s;
  };

  const exportCsv = () => {
    if (!lastResults.length) return;
    const headers = ["timestamp", "server", "channel_id", "author", "content", "link"];
    const rows = [headers.join(",")];
    for (const m of lastResults) {
      // This URL is written into the CSV for the user to click later, not fetched here
      const link = `https://discord.com/channels/${m._guildId}/${m.channel_id}/${m.id}`;
      rows.push([
        csvEscape(m.timestamp),
        csvEscape(m._guildName),
        csvEscape(m.channel_id),
        csvEscape(m.author?.username || ""),
        csvEscape(m.content || ""),
        csvEscape(link),
      ].join(","));
    }
    const csv = "\ufeff" + rows.join("\r\n");
    const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    const safeQuery = currentQuery.replace(/[^a-z0-9]+/gi, "_").slice(0, 40) || "search";
    a.href = url;
    a.download = `discord_search_${safeQuery}.csv`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    setTimeout(() => URL.revokeObjectURL(url), 1000);
  };

  overlay.querySelector("#xsearch-go").onclick = () => {
    currentQuery = overlay.querySelector("#xsearch-q").value.trim();
    if (!currentQuery) return;
    search(currentQuery);
  };
  overlay.querySelector("#xsearch-stop").onclick = () => { stopRequested = true; };
  overlay.querySelector("#xsearch-export").onclick = exportCsv;
  overlay.querySelector("#xsearch-q").addEventListener("keydown", e => {
    if (e.key === "Enter" && !overlay.querySelector("#xsearch-go").disabled) {
      overlay.querySelector("#xsearch-go").click();
    }
  });
})();

(() => {
  // ================================================================
  // FEATURE 4: Fetch + XHR interception for auth token and API version
  // ================================================================
  // Patch fetch and XHR immediately so we passively capture the
  // Authorization header and API version from any outgoing Discord API
  // request. Both patches are fully transparent and forward all calls to
  // the originals. Restores itself once both values are captured.
  let _capturedToken = null;
  let _capturedApiVersion = null;
  // Stealth headers captured from the real client: we replay these exact values
  // so our search requests are header-by-header indistinguishable from Discord's
  // own UI. Without these, requests look automation-shaped (no super-properties,
  // no locale/timezone) which is one of the easier patterns to flag.
  const _capturedHeaders = {};
  // Header allowlist: only the metadata-shaped headers Discord sends from the
  // client. We deliberately exclude content-length, content-type, cookie (the
  // browser handles cookie automatically with credentials:include), referer,
  // user-agent (the browser sets it), and any cache-control.
  const _stealthHeaderAllow = new Set([
    "x-super-properties",
    "x-discord-locale",
    "x-discord-timezone",
    "x-debug-options",
    "x-discord-react-locale",
  ]);
  const _origFetch = window.fetch;
  const _origXHROpen = XMLHttpRequest.prototype.open;
  const _origXHRSetHeader = XMLHttpRequest.prototype.setRequestHeader;
  const _xhrUrls = new WeakMap();

  const _restoreNet = () => {
    window.fetch = _origFetch;
    XMLHttpRequest.prototype.open = _origXHROpen;
    XMLHttpRequest.prototype.setRequestHeader = _origXHRSetHeader;
  };

  const _maybeRestore = () => {
    // Stop intercepting once we have token + version + at least one stealth
    // header. Keeps the patch surface tiny so we do not accidentally affect
    // Discord's own performance.
    if (_capturedToken && _capturedApiVersion && Object.keys(_capturedHeaders).length > 0) {
      _restoreNet();
    }
  };

  const _saveHeader = (name, value) => {
    if (!name || value == null) return;
    const lower = String(name).toLowerCase();
    if (!_stealthHeaderAllow.has(lower)) return;
    if (!_capturedHeaders[lower]) _capturedHeaders[lower] = String(value);
  };

  const _netCapture = (urlStr, authHeader, headerEntries) => {
    if (!urlStr || !urlStr.includes("discord.com/api/")) return;
    if (!_capturedApiVersion) {
      const m = urlStr.match(/discord(?:app)?\.com\/api\/(v\d+)\//);
      if (m) _capturedApiVersion = m[1];
    }
    if (!_capturedToken && typeof authHeader === "string" && authHeader.length > 30) {
      // Strip optional "Bearer " prefix; Discord user tokens are sent raw,
      // but bot/oauth flows include the prefix. Normalize to whatever Discord
      // currently accepts on the search endpoint (the raw header value).
      _capturedToken = authHeader.replace(/^Bearer\s+/i, "");
    }
    if (headerEntries) {
      for (const [name, value] of headerEntries) _saveHeader(name, value);
    }
    _maybeRestore();
  };

  const _entriesFromInit = (init) => {
    if (!init?.headers) return null;
    if (init.headers instanceof Headers) return [...init.headers.entries()];
    if (Array.isArray(init.headers)) return init.headers;
    return Object.entries(init.headers);
  };

  window.fetch = function(url, init) {
    try {
      const urlStr = String(typeof url === "string" ? url : (url?.url ?? url));
      const entries = _entriesFromInit(init);
      let auth;
      if (entries) {
        for (const [n, v] of entries) {
          if (String(n).toLowerCase() === "authorization") { auth = v; break; }
        }
      }
      _netCapture(urlStr, auth, entries);
    } catch (e) {}
    return _origFetch.apply(this, arguments);
  };

  XMLHttpRequest.prototype.open = function(method, url) {
    try { _xhrUrls.set(this, String(url)); } catch (e) {}
    return _origXHROpen.apply(this, arguments);
  };

  XMLHttpRequest.prototype.setRequestHeader = function(name, value) {
    try {
      const url = _xhrUrls.get(this) ?? "";
      if (String(name).toLowerCase() === "authorization") {
        _netCapture(url, value, null);
      } else if (url.includes("discord.com/api/")) {
        _saveHeader(name, value);
        _maybeRestore();
      }
    } catch (e) {}
    return _origXHRSetHeader.apply(this, arguments);
  };

  // ================================================================
  // Webpack chunk discovery: try the known name first, then scan window
  // for any property matching the webpackChunk* pattern in case Discord
  // renames the app bundle
  const getWebpackChunk = () => {
    if (window.webpackChunkdiscord_app) return window.webpackChunkdiscord_app;
    for (const key of Object.getOwnPropertyNames(window)) {
      try {
        if (key.startsWith("webpackChunk") && Array.isArray(window[key]) && window[key].push) {
          return window[key];
        }
      } catch (e) {}
    }
    return null;
  };

  let _wpRequire = null;
  // Webpack chunk IDs are tracked in installedChunks; using Symbol() is fragile
  // because newer webpack builds key chunks via Map keys and may type-check.
  // Use a stable unique string ID instead so the push registers cleanly.
  const _wpProbeId = "_xsearch_probe_" + Math.random().toString(36).slice(2);
  const wp = () => {
    if (_wpRequire) return _wpRequire;
    const chunk = getWebpackChunk();
    if (!chunk) return null;
    try {
      _wpRequire = chunk.push([[_wpProbeId], {}, r => r]);
      chunk.pop();
    } catch (e) {
      // Some builds reject empty modules; try again with a no-op module factory
      try {
        _wpRequire = chunk.push([[_wpProbeId + "_2"], { [_wpProbeId]: () => {} }, r => r]);
        chunk.pop();
      } catch (e2) {}
    }
    return _wpRequire;
  };

  // Extract all possible export surfaces from a webpack module.
  // Checks the top level, the default export, and all object valued
  // keys so renamed or restructured modules still surface.
  const candidatesOf = (mod) => {
    if (!mod?.exports) return [];
    const seen = new Set();
    const out = [];
    const add = (v) => {
      if (v && typeof v === "object" && !seen.has(v)) {
        seen.add(v);
        out.push(v);
      }
    };
    add(mod.exports);
    try {
      if (mod.exports.default) add(mod.exports.default);
      for (const k of Object.keys(mod.exports)) {
        add(mod.exports[k]);
      }
    } catch (e) {}
    return out;
  };

  // Discord's i18n system uses Proxy objects where every property access
  // returns a locale message getter function. These proxies match filters
  // for transitionTo, getToken, getGuilds, etc. because they respond to
  // ANY property name with a function. Detect them by probing a property
  // that could never exist on a real module: if it returns a function,
  // the object is an i18n proxy and should be skipped.
  const isI18nProxy = (obj) => {
    try {
      return typeof obj["__xsearch_i18n_probe__"] === "function";
    } catch (e) {
      return false;
    }
  };

  // Build and cache a flat list of every exported object across all
  // webpack modules, excluding i18n proxies. Cached because the module
  // graph is static at runtime.
  const _buildCandidates = () => {
    const wpRequire = wp();
    if (!wpRequire || !wpRequire.c) return [];
    const out = [];
    for (const id in wpRequire.c) {
      try {
        for (const exp of candidatesOf(wpRequire.c[id])) {
          if (!isI18nProxy(exp)) out.push(exp);
        }
      } catch (e) {}
    }
    return out;
  };
  const _candidates = _buildCandidates();

  // ================================================================
  // FEATURE 1: Dynamic API base URL detection
  // ================================================================
  // Detect the live API version instead of hardcoding v9. Priority order:
  // 1. Already captured from a live request (most authoritative)
  // 2. window.GLOBAL_ENV.API_ENDPOINT exposed by Discord's client build
  // 3. Webpack module export containing an API URL string with a version
  // 4. Hardcoded fallback
  const getApiBase = () => {
    if (_capturedApiVersion) return `https://discord.com/api/${_capturedApiVersion}`;
    try {
      const env = window.GLOBAL_ENV;
      if (env?.API_ENDPOINT) {
        const ep = env.API_ENDPOINT.replace(/\/$/, "");
        if (/\/v\d+$/.test(ep)) return ep;
        if (env.API_VERSION != null) return `${ep}/v${env.API_VERSION}`;
      }
    } catch (e) {}
    for (const exp of _candidates) {
      try {
        for (const key of Object.keys(exp)) {
          const val = exp[key];
          if (typeof val === "string" && /^https:\/\/discord(?:app)?\.com\/api\/v\d+$/.test(val))
            return val;
        }
      } catch (e) {}
    }
    return "https://discord.com/api/v9";
  };
  const API_BASE = getApiBase();

  // TOKEN HANDLING: the token is read from Discord's own internal stores
  // (the same way Discord itself reads it) and is ONLY ever sent to
  // discord.com in the authorization header of search requests.
  // It is never logged, stored externally, or transmitted anywhere else.
  // You can verify this by searching this entire script for "token":
  // it only appears in this function, the variable assignment below,
  // and the fetch() call in the search loop.
  const jwtLike = /^[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{4,}\.[A-Za-z0-9_-]{10,}$/;
  const isTokenLike = (t) => typeof t === "string" && t.length > 30 && jwtLike.test(t);

  const getToken = () => {
    // Strategy 1: network capture (set by the fetch/XHR patch above; most resilient)
    if (_capturedToken) return _capturedToken;
    // Strategy 2: read from Discord's own webpack auth store (kept in memory)
    for (const exp of _candidates) {
      try {
        if (typeof exp.getToken !== "function") continue;
        const t = exp.getToken();
        if (isTokenLike(t)) return t;
      } catch (e) {}
    }
    // Strategy 3: read from localStorage (older builds stored it here)
    try {
      const raw = localStorage.getItem("token");
      if (raw) {
        const cleaned = raw.replace(/^"|"$/g, "");
        if (isTokenLike(cleaned)) return cleaned;
      }
    } catch (e) {}
    // Strategy 4 (FEATURE 3): schema-based structural scan
    // Call every zero-argument function on modules that look like store modules
    // (2 to 30 methods, to avoid large dispatch modules with side effects).
    // Restrict to functions declaring zero formal parameters AND whose name is
    // either getter-shaped (get/is/has/fetch prefix) or single-letter (likely
    // minified) to minimize the chance of triggering side-effectful dispatchers.
    // A JWT-shaped return value is highly distinctive, so false positives are rare.
    const safeFnNameRe = /^(?:get|is|has|fetch|read|peek)|^[a-zA-Z]$/;
    for (const exp of _candidates) {
      try {
        const fns = Object.keys(exp).filter(k =>
          typeof exp[k] === "function"
          && exp[k].length === 0
          && safeFnNameRe.test(k)
        );
        if (fns.length < 1 || fns.length > 30) continue;
        for (const key of fns) {
          const t = exp[key]();
          if (isTokenLike(t)) return t;
        }
      } catch (e) {}
    }
    return null;
  };

  // Guilds: validates that the returned value is a map of guild-shaped objects
  // using guild-specific fields (features, roles, ownerId) so other id/name maps
  // from unrelated modules are not mistaken for guild data.
  const looksLikeGuild = (v) =>
    v && typeof v.id === "string" && typeof v.name === "string"
    && (Array.isArray(v.features) || Array.isArray(v.roles) || v.ownerId != null);

  const isGuildMap = (obj) => {
    if (!obj || typeof obj !== "object" || Array.isArray(obj)) return false;
    const vals = Object.values(obj);
    if (vals.length === 0) return false;
    return vals.slice(0, Math.min(3, vals.length)).every(looksLikeGuild);
  };

  const extractGuildList = (obj) =>
    Object.values(obj)
      .filter(v => v?.id && v?.name)
      .map(v => ({ id: v.id, name: v.name }));

  const getGuilds = () => {
    // Strategy 1: name-based (fast path)
    for (const exp of _candidates) {
      try {
        if (typeof exp.getGuilds !== "function") continue;
        const g = exp.getGuilds();
        if (isGuildMap(g)) return extractGuildList(g);
      } catch (e) {}
    }
    // Strategy 2 (FEATURE 3): schema-based structural scan
    // Only consider zero-argument functions whose name is getter-shaped or
    // single-letter (likely minified). This avoids invoking action creators
    // like leaveGuild, deleteGuild, joinGuild which take args but might also
    // exist on the same module surface.
    const safeFnNameRe = /^(?:get|is|has|fetch|read|peek)|^[a-zA-Z]$/;
    for (const exp of _candidates) {
      try {
        for (const key of Object.keys(exp)) {
          const fn = exp[key];
          if (typeof fn !== "function" || fn.length !== 0) continue;
          if (!safeFnNameRe.test(key)) continue;
          const result = fn.call(exp);
          if (isGuildMap(result)) return extractGuildList(result);
        }
      } catch (e) {}
    }
    return [];
  };

  // Generic module finder for navigation lookups
  const findModule = (filter) => {
    for (const exp of _candidates) {
      try { if (filter(exp)) return exp; } catch (e) {}
    }
    return null;
  };

  // Navigation: three fallback strategies so renames of any single
  // internal module do not break message jumping.
  // Strategy 1: transitionTo (Router module, most common)
  // Strategy 2: selectChannel (ChannelActions module)
  // Strategy 3: history.pushState + popstate (no webpack module needed)
  const getTransitionTo = () => {
    // Strategy 1: name-based with paired method verification
    const mod =
      findModule(m => typeof m.transitionTo === "function" && typeof m.replaceWith === "function")
      ?? findModule(m => typeof m.transitionTo === "function");
    if (mod) return mod.transitionTo.bind(mod);
    // Strategy 2 (FEATURE 3): schema-based router detection
    // React Router exposes a history object with push/replace and a location
    // with pathname. This shape is distinctive enough to match without names.
    const routerMod = findModule(m =>
      m?.history && typeof m.history.push === "function"
      && typeof m.history.replace === "function"
      && m?.location && typeof m.location.pathname === "string"
    );
    if (routerMod) return (path) => routerMod.history.push(path);
    return null;
  };

  const getSelectChannel = () => {
    // Strategy 1: name-based with paired method verification
    const mod =
      findModule(m => typeof m.selectChannel === "function" && typeof m.selectVoiceChannel === "function")
      ?? findModule(m => typeof m.selectChannel === "function");
    if (mod) return mod.selectChannel.bind(mod);
    // Strategy 2 (FEATURE 3): find a channel-actions module by characteristic
    // sibling methods (selectVoiceChannel, selectPrivateChannel) even when
    // selectChannel itself has been renamed by minification.
    const chanMod = findModule(m =>
      typeof m.selectVoiceChannel === "function" || typeof m.selectPrivateChannel === "function"
    );
    if (chanMod) {
      const fn = chanMod.selectChannel ?? chanMod.selectPrivateChannel ?? chanMod.selectVoiceChannel;
      if (typeof fn === "function") return fn.bind(chanMod);
    }
    return null;
  };

  // Discord-native dispatcher fallback. Discord's internal navigation is
  // ultimately driven by Flux actions, so dispatching CHANNEL_SELECT directly
  // works even when both Router and ChannelActions modules have been renamed.
  const getFluxDispatcher = () =>
    findModule(m =>
      typeof m.dispatch === "function"
      && (typeof m.subscribe === "function" || typeof m._subscriptions === "object")
      && (Array.isArray(m._actionHandlers?._orderedActionHandlers)
          || typeof m.register === "function"
          || typeof m.wait === "function")
    );

  const transitionTo = getTransitionTo();
  const selectChannel = getSelectChannel();
  const fluxDispatcher = getFluxDispatcher();

  const jumpToMessage = (guildId, channelId, messageId) => {
    const path = `/channels/${guildId}/${channelId}/${messageId}`;
    try { if (transitionTo) { transitionTo(path); return; } } catch (e) {}
    try { if (selectChannel) { selectChannel({ guildId, channelId, messageId }); return; } } catch (e) {}
    try {
      if (fluxDispatcher) {
        fluxDispatcher.dispatch({ type: "CHANNEL_SELECT", guildId, channelId, messageId });
        return;
      }
    } catch (e) {}
    // Last-resort fallback: history manipulation. Discord may not pick this up
    // if its router does not listen on popstate, but it costs nothing to try.
    try {
      window.history.pushState({}, "", path);
      window.dispatchEvent(new PopStateEvent("popstate"));
    } catch (e) { console.error("[xsearch] no navigation method worked", e); }
  };

  // Init
  // Token is stored in a local variable scoped to this IIFE. It never
  // leaves this closure and is only used in the authorization header
  // of fetch() calls to discord.com (see the search function below).
  let token = getToken();

  const guilds = getGuilds().sort((a, b) => a.name.localeCompare(b.name));
  if (guilds.length === 0) { console.error("[xsearch] Found 0 guilds."); return; }
  if (!token) console.warn("[xsearch] Token not found via webpack; will retry at search time via network capture.");
  console.log(`[xsearch] Ready: ${guilds.length} guilds, nav=${!!(transitionTo || selectChannel || fluxDispatcher)}, api=${API_BASE}, stealthHeaders=${Object.keys(_capturedHeaders).length}`);

  // ================================================================
  // FEATURE 5: CSP-resilient style injection
  // ================================================================

  document.body.appendChild(overlay);

  // Restore previous overlay position from a prior session, clamped to the
  // current viewport so a smaller window cannot orphan the panel offscreen.
  const POS_KEY = "xsearch_pos_v1";
  try {
    const saved = JSON.parse(localStorage.getItem(POS_KEY) || "null");
    if (saved && Number.isFinite(saved.left) && Number.isFinite(saved.top)) {
      const maxLeft = Math.max(0, window.innerWidth - 200);
      const maxTop = Math.max(0, window.innerHeight - 100);
      overlay.style.left = Math.min(maxLeft, Math.max(0, saved.left)) + "px";
      overlay.style.top = Math.min(maxTop, Math.max(0, saved.top)) + "px";
      overlay.style.right = "auto";
    }
  } catch (e) {}
  const savePos = () => {
    try {
      const r = overlay.getBoundingClientRect();
      localStorage.setItem(POS_KEY, JSON.stringify({ left: r.left, top: r.top }));
    } catch (e) {}
  };

  // Global keyboard handler for ESC to close. Stored so we can detach on close.
  const onKeyDown = (e) => {
    if (e.key === "Escape" && document.body.contains(overlay)) {
      // Only close if focus is in the overlay or nothing else is using ESC.
      if (overlay.contains(document.activeElement) || document.activeElement === document.body) {
        closeOverlay();
      }
    }
  };
  document.addEventListener("keydown", onKeyDown);

  const closeOverlay = () => {
    stopRequested = true;
    document.removeEventListener("keydown", onKeyDown);
    _removeStyles?.();
    overlay.remove();
  };

  // Drag handling
  const header = overlay.querySelector("#xsearch-header");
  let dragging = false, dx = 0, dy = 0, dragMoved = false;
  header.addEventListener("mousedown", e => {
    if (e.target.tagName === "BUTTON") return;
    dragging = true;
    dragMoved = false;
    const rect = overlay.getBoundingClientRect();
    dx = e.clientX - rect.left;
    dy = e.clientY - rect.top;
    e.preventDefault();
  });
  document.addEventListener("mousemove", e => {
    if (!dragging) return;
    dragMoved = true;
    overlay.style.left = (e.clientX - dx) + "px";
    overlay.style.top = (e.clientY - dy) + "px";
    overlay.style.right = "auto";
  });
  document.addEventListener("mouseup", () => {
    if (dragging && dragMoved) savePos();
    dragging = false;
  });

  header.addEventListener("click", e => {
    if (e.target.tagName === "BUTTON") return;
    if (dragMoved) return;
    overlay.classList.toggle("expanded");
    overlay.querySelector("#xsearch-toggle").textContent = overlay.classList.contains("expanded") ? "Hide" : "Show";
  });

  overlay.querySelector("#xsearch-toggle").onclick = (e) => {
    e.stopPropagation();
    overlay.classList.toggle("expanded");
    e.target.textContent = overlay.classList.contains("expanded") ? "Hide" : "Show";
  };
  overlay.querySelector("#xsearch-close").onclick = (e) => {
    e.stopPropagation();
    closeOverlay();
  };

  const escapeHtml = (s) => String(s ?? "")
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#39;");

  const serversDiv = overlay.querySelector("#xsearch-servers");
  guilds.forEach(g => {
    const lbl = document.createElement("label");
    const safeName = escapeHtml(g.name);
    const safeId = escapeHtml(g.id);
    lbl.innerHTML = `<input type="checkbox" data-id="${safeId}" checked><span title="${safeName}">${safeName}</span>`;
    serversDiv.appendChild(lbl);
  });

  overlay.querySelector("#xsearch-all").onclick = () => {
    serversDiv.querySelectorAll("input[type=checkbox]").forEach(c => c.checked = true);
  };
  overlay.querySelector("#xsearch-none").onclick = () => {
    serversDiv.querySelectorAll("input[type=checkbox]").forEach(c => c.checked = false);
  };

  // Search state
  let currentQuery = "";
  let lastResults = [];
  let stopRequested = false;
  // Hard kill: set when Discord returns 401 (token revoked) or signals an
  // account-flag condition. When true, every in-flight server worker must
  // exit immediately and the user must re-inject the script.
  let hardKill = false;

  const sleep = ms => new Promise(r => setTimeout(r, ms));
  // Jittered sleep: humans do not click at constant intervals. ±25% noise
  // around the target makes our timing distribution look organic without
  // changing the average request rate.
  const jitter = (ms, pct = 0.25) => {
    const variance = ms * pct;
    return Math.max(0, ms + (Math.random() * 2 - 1) * variance);
  };
  const jitteredSleep = (ms, pct) => sleep(jitter(ms, pct));

  // Conservative defaults that stay well under Discord's per-guild search
  // bucket (roughly 10/5s historically). With a 2.5s mean per page we burst
  // ~2 pages then naturally pace with the bucket. Server delay is the
  // think-pause between switching servers within the same worker thread.
  const PAGE_DELAY = 2500;
  const SERVER_DELAY = 4000;
  const PAGE_SIZE = 25;
  const INDEX_WAIT = 5000;
  const MAX_INDEX_RETRIES = 12;
  // Maximum back-to-back 429s from a single guild before we give up on it.
  // Three in a row indicates a behavior-flag rate limit (much longer than
  // the bucket reset), not a normal bucket exhaustion.
  const MAX_CONSECUTIVE_429 = 3;

  // Highlight matched query terms in result content. Splits on case-insensitive
  // matches and rebuilds the node tree with <mark>-style spans so the user can
  // see WHY each result matched without scanning long messages.
  const highlightInto = (parent, text, query) => {
    parent.textContent = "";
    if (!text) { parent.textContent = "(no text content)"; return; }
    const terms = query.split(/\s+/).filter(t => t.length >= 2);
    if (terms.length === 0) { parent.textContent = text; return; }
    const escaped = terms.map(t => t.replace(/[.*+?^${}()|[\]\\]/g, "\\$&"));
    const re = new RegExp(`(${escaped.join("|")})`, "gi");
    const parts = text.split(re);
    for (const p of parts) {
      if (!p) continue;
      if (re.test(p)) {
        re.lastIndex = 0;
        const span = document.createElement("span");
        span.className = "hl";
        span.textContent = p;
        parent.appendChild(span);
      } else {
        parent.appendChild(document.createTextNode(p));
      }
      re.lastIndex = 0;
    }
  };

  const renderResult = (m) => {
    const results = overlay.querySelector("#xsearch-results");
    const div = document.createElement("div");
    div.className = "msg";
    const date = new Date(m.timestamp).toLocaleString();
    const meta = document.createElement("div");
    meta.className = "meta";
    meta.textContent = `${m._guildName} #${m.channel_id} ${date} ${m.author?.username || "?"}`;
    const content = document.createElement("div");
    content.className = "content";
    highlightInto(content, m.content || "", currentQuery);
    div.appendChild(meta);
    div.appendChild(content);
    div.onclick = () => jumpToMessage(m._guildId, m.channel_id, m.id);
    results.appendChild(div);
  };

  // Discord caps the search 'offset' parameter at 5000. Beyond that we must
  // switch to snowflake cursor pagination via 'max_id' (results older than
  // the given message id) and reset offset to 0.
  const OFFSET_CAP = 5000;

  // Per-guild rate limit tracker. Each guild has its own search bucket on
  // Discord's side; we track its known reset window to skip pacing delays
  // when we have plenty of quota left, and add them back when we are close
  // to the limit. This is what gives us speed safely.
  const _bucketState = new Map(); // guildId -> { remaining, resetAt }

  const updateBucket = (guildId, res) => {
    try {
      const remaining = parseInt(res.headers.get("X-RateLimit-Remaining") ?? "", 10);
      const resetAfter = parseFloat(res.headers.get("X-RateLimit-Reset-After") ?? "");
      if (Number.isNaN(remaining) || Number.isNaN(resetAfter)) return;
      _bucketState.set(guildId, {
        remaining,
        resetAt: Date.now() + resetAfter * 1000,
      });
    } catch (e) {}
  };

  // Compute how long we should sleep before the next request to this guild.
  // - If bucket has plenty of headroom (>= 3 calls remaining), tiny jittered pause
  // - If bucket is near-empty (1-2 calls), full PAGE_DELAY
  // - If bucket is exhausted (0), wait for reset + small buffer
  const computePacing = (guildId) => {
    const b = _bucketState.get(guildId);
    if (!b) return jitter(PAGE_DELAY);
    const msUntilReset = Math.max(0, b.resetAt - Date.now());
    if (b.remaining <= 0) return msUntilReset + 250;
    if (b.remaining <= 2) return Math.max(jitter(PAGE_DELAY), msUntilReset / Math.max(1, b.remaining));
    return jitter(800, 0.4); // headroom: short, varied micro-pause
  };

  // Cloudflare/edge sometimes returns an HTML challenge or block page with a
  // 200 or 403 status. Detect by content-type so we do not crash JSON.parse.
  const isJsonResponse = (res) => {
    const ct = res.headers.get("content-type") || "";
    return ct.includes("application/json");
  };

  // Build the headers for a single search request. Replays any stealth
  // headers we captured from the real client so requests are header-shape
  // identical to Discord's UI traffic.
  const buildSearchHeaders = () => {
    const h = { Authorization: token };
    for (const [name, value] of Object.entries(_capturedHeaders)) {
      h[name] = value;
    }
    return h;
  };

  // Concurrency runner: process items with at most `concurrency` in flight.
  // Workers pull from a shared queue so a fast server does not idle while a
  // slow one is still going.
  const runWithConcurrency = async (items, fn, concurrency) => {
    const queue = items.slice();
    const workerCount = Math.max(1, Math.min(concurrency, queue.length));
    const workers = [];
    for (let i = 0; i < workerCount; i++) {
      workers.push((async () => {
        while (queue.length) {
          if (stopRequested || hardKill) return;
          const item = queue.shift();
          try { await fn(item); } catch (e) { console.error("[xsearch] worker", e); }
        }
      })());
    }
    await Promise.all(workers);
  };

  const searchOneGuild = async (g, query, onPage) => {
    let offset = 0;
    let maxIdCursor = null;
    let oldestIdSeenInBucket = null;
    let serverTotal = null;
    let serverPage = 1;
    let serverPages = 1;
    let indexRetries = 0;
    let consecutive429 = 0;
    const found = [];

    while (true) {
      if (stopRequested || hardKill) return found;

      try {
        // include_nsfw=true so age-gated channels are not silently excluded.
        const params = new URLSearchParams({ content: query, include_nsfw: "true" });
        if (maxIdCursor) params.set("max_id", maxIdCursor);
        else params.set("offset", String(offset));
        const url = `${API_BASE}/guilds/${g.id}/messages/search?${params.toString()}`;
        const res = await fetch(url, {
          headers: buildSearchHeaders(),
          credentials: "include",
        });

        // Hard auth failure: token revoked or invalidated. Stop the entire run.
        if (res.status === 401) {
          hardKill = true;
          console.error("[xsearch] 401 unauthorized - token rejected, halting all searches");
          return found;
        }

        // 403 with code 40002 indicates an account-action flag; halt.
        if (res.status === 403) {
          let bodyCode = null;
          try { if (isJsonResponse(res)) bodyCode = (await res.json())?.code; } catch (e) {}
          if (bodyCode === 40002) {
            hardKill = true;
            console.error("[xsearch] 403 code 40002 - account action required, halting");
            return found;
          }
          console.warn(`[xsearch] ${g.name} returned 403, skipping`);
          return found;
        }

        if (res.status === 429) {
          consecutive429++;
          let retryMs = 20000;
          let isGlobal = false;
          try {
            const headerVal = res.headers.get("Retry-After");
            if (headerVal != null) retryMs = parseFloat(headerVal) * 1000;
            if (isJsonResponse(res)) {
              const body = await res.json();
              if (typeof body.retry_after === "number") retryMs = body.retry_after * 1000;
              if (body.global === true) isGlobal = true;
            }
          } catch (e) {}

          // A global rate limit means we should pause ALL servers, not just this one.
          // We do this by setting a shared sleep barrier; here we just sleep extra long.
          if (isGlobal) {
            console.warn(`[xsearch] GLOBAL rate limit hit, pausing ${Math.ceil(retryMs / 1000)}s`);
            retryMs = Math.max(retryMs, 30000);
          }

          // Back off exponentially on consecutive 429s for the same guild.
          // Three in a row means we are likely behavior-flagged; back off hard
          // then give up on this guild for the run.
          if (consecutive429 >= MAX_CONSECUTIVE_429) {
            console.warn(`[xsearch] ${g.name} hit ${consecutive429} consecutive 429s, skipping`);
            await sleep(retryMs * 2);
            return found;
          }
          retryMs = Math.max(retryMs, retryMs * Math.pow(1.5, consecutive429 - 1));

          const totalSecs = Math.max(1, Math.ceil(retryMs / 1000));
          for (let remaining = totalSecs; remaining > 0; remaining--) {
            if (stopRequested || hardKill) return found;
            onPage?.({ kind: "ratelimit", remaining, guild: g.name });
            await sleep(1000);
          }
          continue;
        }
        consecutive429 = 0;
        updateBucket(g.id, res);

        if (res.status === 202) {
          indexRetries++;
          if (indexRetries > MAX_INDEX_RETRIES) {
            console.warn(`[xsearch] ${g.name} still indexing after ${MAX_INDEX_RETRIES} retries, skipping`);
            return found;
          }
          onPage?.({ kind: "indexing", attempt: indexRetries, guild: g.name });
          await sleep(INDEX_WAIT);
          continue;
        }

        if (!res.ok) {
          console.warn(`[xsearch] ${g.name} returned ${res.status}`);
          return found;
        }

        if (!isJsonResponse(res)) {
          console.warn(`[xsearch] ${g.name} returned non-JSON response, skipping`);
          return found;
        }

        const data = await res.json();

        const bodyIndicatesIndexing = data.indexing
          || (offset === 0 && !maxIdCursor && data.total_results === 0
              && data.analytics_id && !(data.messages?.length));
        if (bodyIndicatesIndexing) {
          indexRetries++;
          if (indexRetries > MAX_INDEX_RETRIES) {
            console.warn(`[xsearch] ${g.name} still indexing after ${MAX_INDEX_RETRIES} retries, skipping`);
            return found;
          }
          onPage?.({ kind: "indexing", attempt: indexRetries, guild: g.name });
          await sleep(INDEX_WAIT);
          continue;
        }

        indexRetries = 0;

        if (serverTotal == null) {
          serverTotal = data.total_results || 0;
          serverPages = Math.max(1, Math.ceil(serverTotal / PAGE_SIZE));
        }
        const groups = data.messages || [];
        if (groups.length === 0) return found;

        for (const group of groups) {
          const hit = group.find(m => m.hit) || group[0];
          if (hit) {
            const enriched = { ...hit, _guildId: g.id, _guildName: g.name };
            found.push(enriched);
            if (!oldestIdSeenInBucket || BigInt(hit.id) < BigInt(oldestIdSeenInBucket)) {
              oldestIdSeenInBucket = hit.id;
            }
          }
        }

        onPage?.({
          kind: "page",
          guild: g.name,
          page: serverPage,
          totalPages: serverPages,
          serverTotal,
          fraction: serverPage / serverPages,
          newResults: found.length,
        });

        // Pagination: prefer offset within a bucket; flip to max_id cursor
        // once we approach Discord's hard 5000 offset cap.
        if (!maxIdCursor) {
          offset += PAGE_SIZE;
          if (groups.length < PAGE_SIZE) return found;
          if (offset >= serverTotal) return found;
          if (offset >= OFFSET_CAP) {
            if (!oldestIdSeenInBucket) return found;
            maxIdCursor = oldestIdSeenInBucket;
            oldestIdSeenInBucket = null;
            offset = 0;
          }
        } else {
          if (groups.length < PAGE_SIZE) return found;
          if (!oldestIdSeenInBucket) return found;
          if (oldestIdSeenInBucket === maxIdCursor) return found;
          maxIdCursor = oldestIdSeenInBucket;
          oldestIdSeenInBucket = null;
        }
        serverPage++;

        // Adaptive pacing: pull the wait time from the live bucket state.
        await sleep(computePacing(g.id));
      } catch (e) {
        console.error(`[xsearch] ${g.name}`, e);
        return found;
      }
    }
  };

  const search = async (query) => {
    // Re-check token here; it may have been captured from the network after init
    if (!token) token = getToken();

    const checked = [...serversDiv.querySelectorAll("input:checked")].map(i => ({
      id: i.dataset.id,
      name: i.parentElement.querySelector("span").textContent.trim()
    }));

    const status = overlay.querySelector("#xsearch-status");
    const results = overlay.querySelector("#xsearch-results");
    const progressWrap = overlay.querySelector("#xsearch-progress-wrap");
    const progressBar = overlay.querySelector("#xsearch-progress-bar");
    const exportBtn = overlay.querySelector("#xsearch-export");
    const goBtn = overlay.querySelector("#xsearch-go");
    const stopBtn = overlay.querySelector("#xsearch-stop");
    const concurrencyInput = overlay.querySelector("#xsearch-concurrency");
    const concurrency = Math.max(1, Math.min(5, parseInt(concurrencyInput?.value ?? "2", 10) || 2));

    results.innerHTML = "";
    progressWrap.style.display = "block";
    progressBar.style.width = "0%";
    exportBtn.disabled = true;
    goBtn.disabled = true;
    stopBtn.disabled = false;
    stopRequested = false;
    hardKill = false;
    lastResults = [];

    if (!token) {
      status.textContent = "Error: could not retrieve auth token. Open any channel in Discord then click Search again.";
      goBtn.disabled = false;
      stopBtn.disabled = true;
      progressWrap.style.display = "none";
      return;
    }
    if (checked.length === 0) {
      status.textContent = "Select at least one server.";
      goBtn.disabled = false;
      stopBtn.disabled = true;
      progressWrap.style.display = "none";
      return;
    }

    const all = [];
    const total = checked.length;
    let completedServers = 0;
    const serverProgress = new Map(); // guildId -> [0..1]

    const updateProgress = () => {
      let sum = completedServers;
      for (const f of serverProgress.values()) sum += Math.min(1, f);
      const pct = (sum / total) * 100;
      progressBar.style.width = `${Math.min(100, pct)}%`;
    };

    const renderStatus = (extra = "") => {
      const inFlight = serverProgress.size;
      status.textContent =
        `${completedServers}/${total} servers done, ${inFlight} active, ${all.length} results` +
        (extra ? ` - ${extra}` : "");
    };

    renderStatus();

    await runWithConcurrency(checked, async (g) => {
      if (stopRequested || hardKill) return;
      serverProgress.set(g.id, 0);
      renderStatus(`starting ${g.name}`);
      // Small randomized stagger so we never start parallel workers in lockstep
      await sleep(jitter(300, 0.7));

      const found = await searchOneGuild(g, query, (evt) => {
        if (evt.kind === "page") {
          serverProgress.set(g.id, evt.fraction);
          updateProgress();
          renderStatus(`${evt.guild} page ${evt.page}/${evt.totalPages}`);
        } else if (evt.kind === "ratelimit") {
          renderStatus(`${evt.guild} rate-limited, waiting ${evt.remaining}s`);
        } else if (evt.kind === "indexing") {
          renderStatus(`${evt.guild} indexing (attempt ${evt.attempt})`);
        }
      });

      for (const m of found) all.push(m);
      serverProgress.delete(g.id);
      completedServers++;
      updateProgress();
      renderStatus(`finished ${g.name}`);

      // Inter-server think pause within a worker, jittered. Skipped on the
      // last item per worker (queue is empty) by virtue of the loop exit.
      if (!stopRequested && !hardKill) await jitteredSleep(SERVER_DELAY);
    }, concurrency);

    // Deduplicate by message id (cursor + offset modes can occasionally overlap).
    const seenIds = new Set();
    const deduped = [];
    for (const m of all) {
      if (seenIds.has(m.id)) continue;
      seenIds.add(m.id);
      deduped.push(m);
    }
    deduped.sort((a, b) => new Date(a.timestamp) - new Date(b.timestamp));
    lastResults = deduped;

    let finalNote = "";
    if (hardKill) finalNote = " (HALTED: token rejected or account flagged - re-inject script)";
    else if (stopRequested) finalNote = " (stopped early)";
    status.textContent = `${deduped.length} total results across ${total} servers${finalNote}`;
    progressWrap.style.display = "none";
    progressBar.style.width = "100%";
    exportBtn.disabled = deduped.length === 0;
    goBtn.disabled = false;
    stopBtn.disabled = true;

    // Render in chunks via requestAnimationFrame so 5000+ results do not lock
    // the UI thread for seconds. Each frame renders up to 50 results.
    let i = 0;
    const renderChunk = () => {
      const end = Math.min(deduped.length, i + 50);
      for (; i < end; i++) renderResult(deduped[i]);
      if (i < deduped.length) requestAnimationFrame(renderChunk);
    };
    renderChunk();
  };

  // CSV export (local file download only, no network requests)
  const csvEscape = (v) => {
    if (v == null) return "";
    const s = String(v);
    if (/[",\n\r]/.test(s)) return `"${s.replace(/"/g, '""')}"`;
    return s;
  };

  const exportCsv = () => {
    if (!lastResults.length) return;
    const headers = ["timestamp", "server", "channel_id", "author", "content", "link"];
    const rows = [headers.join(",")];
    for (const m of lastResults) {
      // This URL is written into the CSV for the user to click later, not fetched here
      const link = `https://discord.com/channels/${m._guildId}/${m.channel_id}/${m.id}`;
      rows.push([
        csvEscape(m.timestamp),
        csvEscape(m._guildName),
        csvEscape(m.channel_id),
        csvEscape(m.author?.username || ""),
        csvEscape(m.content || ""),
        csvEscape(link),
      ].join(","));
    }
    const csv = "\ufeff" + rows.join("\r\n");
    const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    const safeQuery = currentQuery.replace(/[^a-z0-9]+/gi, "_").slice(0, 40) || "search";
    a.href = url;
    a.download = `discord_search_${safeQuery}.csv`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    setTimeout(() => URL.revokeObjectURL(url), 1000);
  };

  overlay.querySelector("#xsearch-go").onclick = () => {
    const raw = overlay.querySelector("#xsearch-q").value.trim();
    const status = overlay.querySelector("#xsearch-status");
    if (!raw) {
      status.textContent = "Enter a search query first.";
      overlay.querySelector("#xsearch-q").focus();
      return;
    }
    if (raw.length < 2) {
      status.textContent = "Query must be at least 2 characters.";
      return;
    }
    currentQuery = raw;
    search(currentQuery);
  };
  overlay.querySelector("#xsearch-stop").onclick = () => {
    stopRequested = true;
    overlay.querySelector("#xsearch-status").textContent += " - stopping...";
  };
  overlay.querySelector("#xsearch-export").onclick = exportCsv;
  overlay.querySelector("#xsearch-q").addEventListener("keydown", e => {
    if (e.key === "Enter" && !overlay.querySelector("#xsearch-go").disabled) {
      overlay.querySelector("#xsearch-go").click();
    }
  });
})();
